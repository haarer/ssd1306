56 template <class I>
57 void InterfacePCD8544<I>::startBlock(lcduint_t x, lcduint_t y, lcduint_t w)
58 {
59  m_width = w;
60  m_column = x;
61  m_page = y;
62  this->start();
63  spiDataMode(0);
64  if (w == 1) this->send( 0x22 ); else this->send( 0x20 );
65  this->send(0x80 | x);
66  this->send(0x40 | y);
67  spiDataMode(1);
68 }
