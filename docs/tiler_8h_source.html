209  {
210  if ((point.x<0)||(point.y<0) || ((point.y/canvas.height())>=NE_MAX_TILES_NUM)) return;
211  m_refreshFlags[(point.y/canvas.height())] |= (1<<(point.x/canvas.width()));

218  void refresh(lcdint_t x1, lcdint_t y1, lcdint_t x2, lcdint_t y2) __attribute__ ((noinline))
219  {
220  if (y2 < 0 || x2 < 0) return;
221  if (y1 < 0) y1 = 0;
222  if (x1 < 0) x1 = 0;
223  y1 = y1/canvas.height();
224  y2 = min((y2/canvas.height()), NE_MAX_TILES_NUM - 1);
225  for (uint8_t y=y1; y<=y2; y++)
226  {
227  for(uint8_t x=x1/canvas.width(); x<=(x2/canvas.width()); x++)
228  {
229  m_refreshFlags[y] |= (1<<x);
230  }
231  }
232  }

463 void NanoEngineTiler<C,D>::displayBuffer()
464 {
465 // printf("--------------\n");
466  for (lcduint_t y = 0; y < m_display.height(); y = y + canvas.height())
467  {
468  uint16_t flag = m_refreshFlags[y/canvas.height()];
469  m_refreshFlags[y/canvas.height()] = 0;
470 // printf("|%d%d%d%d%d%d%d%d|\n", flag & 1, (flag >> 1) & 1, (flag >> 2) & 1, (flag >> 3) & 1, (flag >> 4) & 1, (flag >> 5) & 1,(flag >> 6) & 1,(flag >> 7) & 1 );
471  for (lcduint_t x = 0; x < m_display.width(); x = x + canvas.width())
472  {
473  if (flag & 0x01)
474  {
475  canvas.setOffset(x + offset.x, y + offset.y);
476  if ( m_onDraw == nullptr )
477  {
478  canvas.clear();
479  draw();
480  this->m_display.drawCanvas(x,y,canvas);
481  }
482  else if ( m_onDraw() )
483  {

485  this->m_display.drawCanvas(x,y,canvas);

487  }
488  flag >>=1;
489  }

506  {
